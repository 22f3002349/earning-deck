Code Sample: NPV in Python


def npv(rate, cashflows):
    """
    rate: discount rate as decimal
    cashflows: list, cashflows[0] may be initial outlay
    """
    return sum(cf / ((1 + rate) ** t) for t, cf in enumerate(cashflows))
print(npv(0.12, [-1000, 300, 350, 400, 500]))
